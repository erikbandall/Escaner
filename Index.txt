<script>
  const SUBTRACT = 204 + 47.525;
  const scanHistory = [];
  let scanning = false;
  let cooldown = false;
  let lastCode = '';

  document.getElementById('barcodeInput').addEventListener('keydown', e => {
    if (e.key === 'Enter') processManual();
  });

  function startScanner() {
    const viewport = document.getElementById('scannerViewport');
    viewport.style.display = 'block';

    Quagga.init({
      inputStream: {
        name: 'Live',
        type: 'LiveStream',
        target: viewport,
        constraints: {
          facingMode: 'environment',
          width: { ideal: 1280 },
          height: { ideal: 720 }
        }
      },
      decoder: {
        readers: [
          'code_128_reader',
          'ean_reader',
          'ean_8_reader',
          'code_39_reader',
          'code_93_reader',
          'upc_reader',
          'upc_e_reader',
          'i2of5_reader',
          'codabar_reader'
        ],
        debug: { drawBoundingBox: false, showFrequency: false, drawScanline: false, showPattern: false }
      },
      locate: true
    }, function(err) {
      if (err) {
        viewport.style.display = 'none';
        setStatus('error', 'Error de cámara: ' + (err.message || err));
        return;
      }
      Quagga.start();
      scanning = true;
      document.getElementById('startBtn').style.display = 'none';
      document.getElementById('stopBtn').style.display  = 'flex';
      setStatus('active', 'Cámara activa -- apunta al código de barras');
    });

    Quagga.onDetected(function(result) {
      if (cooldown) return;
      const code = result.codeResult.code;
      if (!code || code === lastCode) return;

      lastCode = code;
      cooldown = true;
      setTimeout(() => { cooldown = false; lastCode = ''; }, 2500);

      processCode(code);
    });
  }

  function stopScanner() {
    if (!scanning) return;
    try { Quagga.stop(); } catch(e) {}
    scanning = false;
    document.getElementById('scannerViewport').style.display = 'none';
    document.getElementById('startBtn').style.display = 'flex';
    document.getElementById('stopBtn').style.display  = 'none';
    setStatus('', 'Cámara detenida');
  }

  function processManual() {
    const val = document.getElementById('barcodeInput').value.trim();
    if (!val) return;
    processCode(val);
    document.getElementById('barcodeInput').value = '';
  }

  function processCode(raw) {
    const digits = raw.replace(/\D/g, '');

    document.getElementById('idleMsg').style.display = 'none';
    document.getElementById('resultContent').style.display = 'block';
    document.getElementById('errorBox').style.display  = 'none';
    document.getElementById('resultBox').style.display  = 'none';

    if (digits.length < 5) {
      showError(`El código "${raw}" tiene solo ${digits.length} dígito(s). Se necesitan al menos 5.`);
      return;
    }

    const last5   = digits.slice(-5);
    const intPart = last5.slice(0, 3);
    const decPart = last5.slice(3, 5);
    const value   = parseFloat(`${intPart}.${decPart}`);
    const result  = value - SUBTRACT;

    document.getElementById('rFull').textContent     = raw;
    document.getElementById('rLast5').textContent    = last5;
    document.getElementById('rInt').textContent      = intPart + ' mm';
    document.getElementById('rDec').textContent      = '.' + decPart;
    document.getElementById('fScanned').textContent  = value.toFixed(2);
    document.getElementById('fResult').textContent   = result.toFixed(3);
    document.getElementById('rFinalValue').innerHTML = `${result.toFixed(3)}<span class="result-unit">mm</span>`;

    const box = document.getElementById('resultBox');
    box.style.display = 'flex';
    // Re-trigger animation
    box.style.animation = 'none';
    box.offsetHeight;
    box.style.animation = '';

    addHistory(raw, value, result);
  }

  function showError(msg) {
    document.getElementById('resultBox').style.display  = 'none';
    document.getElementById('errorBox').style.display   = 'flex';
    document.getElementById('errorMsg').textContent     = msg;
  }

  function setStatus(type, text) {
    const el = document.getElementById('camStatus');
    el.className = 'cam-status ' + type;
    document.getElementById('camStatusText').textContent = text;
  }

  function addHistory(raw, scanned, result) {
    scanHistory.unshift({ raw, scanned, result });
    if (scanHistory.length > 10) scanHistory.pop();
    renderHistory();
  }

  function renderHistory() {
    document.getElementById('historySection').style.display = 'block';
    document.getElementById('historyList').innerHTML = scanHistory.map(item => `
      <div class="history-item">
        <span class="hi-code">${item.raw}</span>
        <span class="hi-scan">${item.scanned.toFixed(2)}</span>
        <span class="hi-res">${item.result.toFixed(3)} <small>mm</small></span>
      </div>
    `).join('');
  }

  function clearHistory() {
    scanHistory.length = 0;
    document.getElementById('historySection').style.display = 'none';
  }
</script>
